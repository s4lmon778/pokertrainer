# PokerBot Arena — Rebuild Implementation Plan

> **For Hermes:** Execute tasks sequentially. Each task is bite-sized. Commit after each.

**Goal:** Rebuild PokerBot Arena as a polished single-player poker game with bot evaluation dashboard — practice poker against AI opponents while measuring bot decision accuracy in real-time.

**Architecture:** React + TypeScript + Vite + Zustand (state) + Tailwind CSS + Recharts (analytics). Single SPA with tab-based navigation. No backend needed — all data local, persisted in localStorage.

**Tech Stack:** React 19, TypeScript, Vite 8, Zustand, Tailwind CSS 3, Recharts, Lucide icons.

---

## Critical Bugs Found in Current Code

1. **`playerAct` doesn't update pot/chips** — folds and checks work, but call/raise never updates `gameState.pot` or player chips
2. **`advancePhase` resets bets but doesn't clear `actedThisRound`** — bots get stuck
3. **`endHand` only pays the winner but doesn't return bets to non-winners** — chips leak
4. **`nextHand` calls `startHand` without resetting handNumber** — hand counter stays at 1
5. **Bot auto-play useEffect has stale closure** — `botAct()` result not properly awaited before advancing
6. **`resolveHand` sets `gameOver` but `endHand` is never called** — stats never recorded
7. **`GameState.winner` type mismatch** — store uses `{playerId, hand}` but `endHand` expects `Player`
8. **No localStorage persistence** — all stats lost on refresh

import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import type { GameState, Player, GamePhase, Card as CardType, HandEvaluation } from '../types/card';
import { createDeck, shuffleDeck } from '../utils/deck';
import { evaluateHand } from '../utils/handEvaluator';
import { botDecision, type BotSettings, createBotSettings, type BotPersonality } from '../utils/botEngine';

interface GameHistoryEntry {
  handNumber: number;
  winner: string;
  winningHand: string;
  potSize: number;
  duration: number;
  startTime: number;
  endTime: number;
  botResult: number;
  bluffs: number;
  mistakes: number;
  humanAccuracy?: number;
}

interface Stats {

  totalWon: number;

interface GameStore {

const DEFAULT_STATS: Stats = {
  totalHands: 0,
  totalWon: 0,
  totalLost: 0,
  biggestWin: 0,
  biggestLoss: 0,
  avgPotSize: 0,
  botSessions: 0,
  botWins: 0,

const DEFAULT_BOT_EVALUATIONS: BotEvaluationEntry[] = [];

export const useGameStore = create<GameStore>()(
  persist(
    (set, get) => ({
      gameState: null,
      gamePhase: 'preflop',
      isPlaying: false,
      isDealing: false,
      botSettings: createBotSettings('balanced'),
      gameHistory: [],
      stats: { ...DEFAULT_STATS },
      currentBankroll: 1000,
      selectedBotPersonality: 'balanced',
      tableSize: 6,
      buyIn: 100,
      blinds: { small: 5, big: 10 },
      showRiskOverlay: true,
      showTableTalk: false,
      botEvaluations: [...DEFAULT_BOT_EVALUATIONS],
      autoPlaySpeed: 800,

      initializeGame: () => {
        const { tableSize, buyIn, blinds } = get();
        const players: Player[] = [];

        players.push({
          id: 'human',
          name: 'You',
          chips: buyIn,
          hand: [],
          bet: 0,
          folded: false,
          isBot: false,
          position: 0,
          totalBetThisRound: 0,
          actedThisRound: false,
        });

        const botNames = ['Vortex', 'Phantom', 'Shadow', 'Ace', 'Titan', 'Nova', 'Blitz', 'Cipher'];
        for (let i = 1; i < tableSize; i++) {
          players.push({
            id: `bot-${i}`,
            name: botNames[(i - 1) % botNames.length],
            chips: buyIn,
            hand: [],
            bet: 0,
            folded: false,
            isBot: true,
            position: i,
            totalBetThisRound: 0,
            actedThisRound: false,
          });
        }

        set({
          gameState: {
            players,
            communityCards: [],
            pot: 0,
            sidePots: [],
            currentPhase: 'preflop',
            dealerPosition: 0,
            currentPlayerIndex: 0,
            currentBet: blinds.big,
            minRaise: blinds.big,
            handNumber: 0,
            deck: [],
            gameOver: false,
          },
          isPlaying: true,
          gamePhase: 'preflop',
        });
      },

      startHand: () => {
        const state = get();
        if (!state.gameState) return;

        const deck = shuffleDeck(createDeck());
        const { blinds, gameState } = state;
        const players: Player[] = gameState.players.map(p => ({
          ...p,
          hand: [] as CardType[],
          bet: 0,
          folded: false,
          totalBetThisRound: 0,
          actedThisRound: false,
          chips: p.chips, // preserve chip count from previous hand
        }));

        const sbIdx = (gameState.dealerPosition + 1) % players.length;
        const bbIdx = (gameState.dealerPosition + 2) % players.length;
        
        // Post blinds
        players[sbIdx].chips = Math.max(0, players[sbIdx].chips - blinds.small);
        players[sbIdx].bet = blinds.small;
        players[sbIdx].totalBetThisRound = blinds.small;
        players[bbIdx].chips = Math.max(0, players[bbIdx].chips - blinds.big);
        players[bbIdx].bet = blinds.big;
        players[bbIdx].totalBetThisRound = blinds.big;

        // Deal hole cards
        for (let i = 0; i < players.length; i++) {
          const c1 = deck.pop() as CardType | undefined;
          const c2 = deck.pop() as CardType | undefined;
          if (c1 && c2) players[i].hand = [c1, c2];
        }

        const totalPot = blinds.small + blinds.big;

        set({
          gameState: {
            ...gameState,
            players,
            communityCards: [],
            pot: totalPot,
            currentPhase: 'preflop',
            dealerPosition: (gameState.dealerPosition + 1) % players.length,
            currentPlayerIndex: (bbIdx + 1) % players.length,
            currentBet: blinds.big,
            minRaise: blinds.big,
            handNumber: gameState.handNumber + 1,
            deck,
            gameOver: false,
          },
          gamePhase: 'preflop',
          isDealing: true,
        });

        setTimeout(() => set({ isDealing: false }), 600);
      },

      dealCommunityCards: (phase: GamePhase) => {
        const state = get();
        if (!state.gameState) return;
        const { deck, communityCards } = state.gameState;
        const cardsToDeal = phase === 'flop' ? 3 : 1;
        const newCommunity = [...communityCards];
        for (let i = 0; i < cardsToDeal; i++) {
          const card = deck.pop() as CardType | undefined;
          if (card) newCommunity.push(card);
        }

        // Reset bets and acted flags for new phase
        const players = state.gameState.players.map(p => ({
          ...p,
          bet: 0,
          totalBetThisRound: 0,
          actedThisRound: false,
        }));

        set({
          gameState: {
            ...state.gameState,
            players,
            communityCards: newCommunity,
            currentPhase: phase,
            currentPlayerIndex: (state.gameState.dealerPosition + 1) % state.gameState.players.length,
            currentBet: 0,
            minRaise: state.blinds.big,
            lastAction: `${phase.charAt(0).toUpperCase() + phase.slice(1)} dealt`,
          },
          gamePhase: phase,
        });
      },

      botAct: async () => {
        const state = get();
        if (!state.gameState) return;

        const player = state.gameState.players[state.gameState.currentPlayerIndex];
        if (!player || player.isBot === false) return;
        if (player.folded) return;

        const delay = state.autoPlaySpeed;
        await new Promise(resolve => setTimeout(resolve, delay));

        const decision = botDecision(state.gameState!, player, state.botSettings);

        const players = state.gameState.players.map(p => ({ ...p }));
        const pIdx = state.gameState.currentPlayerIndex;
        const currentPlayer = { ...players[pIdx] };

        switch (decision.action) {
          case 'fold':
            currentPlayer.folded = true;
            break;
          case 'check':
            break;
          case 'call': {
            const callAmount = Math.min(state.gameState.currentBet - currentPlayer.bet, currentPlayer.chips);
            currentPlayer.chips -= callAmount;
            currentPlayer.bet += callAmount;
            currentPlayer.totalBetThisRound += callAmount;
            break;
          }
          case 'raise': {
            const raiseTotal = decision.amount || state.gameState.currentBet * 2;
            const toAdd = Math.min(raiseTotal - currentPlayer.bet, currentPlayer.chips);
            currentPlayer.chips -= toAdd;
            currentPlayer.bet += toAdd;
            currentPlayer.totalBetThisRound += toAdd;
            break;
          }
        }

        currentPlayer.actedThisRound = true;
        players[pIdx] = currentPlayer;

        const newPot = state.gameState.pot + currentPlayer.totalBetThisRound - 
          (state.gameState.players[pIdx]?.totalBetThisRound || 0);

        set({
          gameState: {
            ...state.gameState,
            players,
            pot: newPot,
            currentBet: decision.action === 'raise' ? (decision.amount || state.gameState.currentBet) : state.gameState.currentBet,
            lastAction: `${currentPlayer.name} ${decision.action}${decision.action === 'raise' ? ` to $${decision.amount}` : ''}${decision.isBluff ? ' (bluff!)' : ''}`,
          },
        });
      },

      playerAct: (action, amount) => {
        const state = get();
        if (!state.gameState) return;
        const players = state.gameState.players.map(p => ({ ...p }));
        const humanIdx = players.findIndex(p => !p.isBot);
        if (humanIdx === -1) return;
        const human = { ...players[humanIdx] };
        const gs = state.gameState;

        switch (action) {
          case 'fold':
            human.folded = true;
            break;
          case 'check':
            break;
          case 'call': {
            const callAmt = Math.min(gs.currentBet - human.bet, human.chips);
            if (callAmt <= 0) {
              // Already called or checked
              human.actedThisRound = true;
              players[humanIdx] = human;
              set({ gameState: { ...gs, players, lastAction: 'You check' } });
              return;
            }
            human.chips -= callAmt;
            human.bet += callAmt;
            human.totalBetThisRound += callAmt;
            break;
          }
          case 'raise': {
            const raiseTotal = amount || Math.min(gs.currentBet * 2, gs.currentBet + human.chips);
            const toAdd = Math.min(raiseTotal - human.bet, human.chips);
            if (toAdd <= 0) {
              human.actedThisRound = true;
              players[humanIdx] = human;
              set({ gameState: { ...gs, players, lastAction: 'You check' } });
              return;
            }
            human.chips -= toAdd;
            human.bet += toAdd;
            human.totalBetThisRound += toAdd;
            break;
          }
        }

        human.actedThisRound = true;
        players[humanIdx] = human;

        const potIncrease = human.totalBetThisRound - (gs.players[humanIdx]?.totalBetThisRound || 0);
        const newPot = gs.pot + Math.max(0, potIncrease);

        set({
          gameState: {
            ...gs,
            players,
            pot: newPot,
            currentBet: action === 'raise' ? human.bet : gs.currentBet,
            lastAction: `You ${action}${action === 'raise' && amount ? ` to $${amount}` : ''}`,
          },
        });
      },

      advancePhase: () => {
        const state = get();
        if (!state.gameState) return;

        const phases: GamePhase[] = ['preflop', 'flop', 'turn', 'river'];
        const currentIdx = phases.indexOf(state.gameState.currentPhase);

        if (currentIdx < phases.length - 1) {
          const nextPhase = phases[currentIdx + 1];
          get().dealCommunityCards(nextPhase);
        }
      },

      resolveHand: () => {
        const state = get();
        if (!state.gameState) return;

        const activePlayers = state.gameState.players.filter(p => !p.folded);
        if (activePlayers.length === 0) {
          // Everyone folded — shouldn't happen, but handle it
          return;
        }

        if (activePlayers.length === 1) {
          // Only one player left — they win by default
          const winner = activePlayers[0];
          set({
            gameState: {
              ...state.gameState,
              gameOver: true,
              lastAction: `${winner.name} wins (everyone folded)!`,
            },
          });
          get().endHand(winner.id);
          return;
        }

        let bestEval: { playerId: string; hand: HandEvaluation } | null = null;

        for (const player of activePlayers) {
          const handEval = evaluateHand(player.hand, state.gameState.communityCards);
          if (!bestEval || handEval.score > bestEval.hand.score) {
            bestEval = { playerId: player.id, hand: handEval };
          }
        }

        if (bestEval) {
          set({
            gameState: {
              ...state.gameState,
              gameOver: true,
              winner: {
                playerId: bestEval.playerId,
                hand: bestEval.hand,
              },
              lastAction: `${bestEval.playerId === 'human' ? 'You' : bestEval.playerId} wins with ${bestEval.hand.description}!`,
            },
          });
          get().endHand(bestEval.playerId);
        }
      },

      endHand: (winnerId: string) => {
        const state = get();
        if (!state.gameState) return;

        const { gameState, stats, currentBankroll } = state;
        const pot = gameState.pot;
        const isHumanWinner = winnerId === 'human';
        const newBankroll = isHumanWinner ? currentBankroll + pot : currentBankroll;
        const botProfit = isHumanWinner ? -pot : pot;

        const newTotalHands = stats.totalHands + 1;
        const newAvgPot = (stats.avgPotSize * stats.totalHands + pot) / newTotalHands;

        const accuracy = calculateHumanAccuracy(state.gameState, state.gameHistory.length + 1);

        const newStats: Stats = {
          ...stats,
          totalHands: newTotalHands,
          totalWon: stats.totalWon + (isHumanWinner ? 1 : 0),
          totalLost: stats.totalLost + (isHumanWinner ? 0 : 1),
          biggestWin: Math.max(stats.biggestWin, isHumanWinner ? pot : 0),
          biggestLoss: Math.min(stats.biggestLoss, isHumanWinner ? 0 : -pot),
          avgPotSize: newAvgPot,
          botSessions: stats.botSessions + 1,
          botWins: stats.botWins + (isHumanWinner ? 0 : 1),
          botLosses: stats.botLosses + (isHumanWinner ? 1 : 0),
          winRate: ((stats.totalWon + (isHumanWinner ? 1 : 0)) / newTotalHands) * 100,
          roi: ((newBankroll - 1000) / 1000) * 100,
          bankrollHistory: [...stats.bankrollHistory, newBankroll],
        };

        const historyEntry: GameHistoryEntry = {
          handNumber: gameState.handNumber,
          winner: gameState.players.find(p => p.id === winnerId)?.name || winnerId,
          winningHand: gameState.winner?.hand.description || '',
          potSize: pot,
          duration: 0,
          startTime: Date.now(),
          endTime: Date.now(),
          botResult: botProfit,
          bluffs: 0,
          mistakes: 0,
          humanAccuracy: accuracy,
        };

        set({
          stats: newStats,
          gameHistory: [...state.gameHistory, historyEntry],
          currentBankroll: newBankroll,
          isPlaying: false,
        });
      },

      updateBotSettings: (settings) => {
        set(state => ({
          botSettings: { ...state.botSettings, ...settings },
        }));
      },

      setPersonality: (personality) => {
        const settings = createBotSettings(personality);
        set({ botSettings: settings, selectedBotPersonality: personality });
      },

      setTableSize: (size) => {
        set({ tableSize: size });
      },

      setBuyIn: (buyIn) => {
        set({ buyIn });
      },

      toggleRiskOverlay: () => {
        set(state => ({ showRiskOverlay: !state.showRiskOverlay }));
      },

      toggleTableTalk: () => {
        set(state => ({ showTableTalk: !state.showTableTalk }));
      },

      resetStats: () => {
        set({ 
          stats: { ...DEFAULT_STATS }, 
          currentBankroll: 1000, 
          gameHistory: [],
          botEvaluations: [...DEFAULT_BOT_EVALUATIONS],
        });
      },

      nextHand: () => {
        const state = get();
        if (!state.isPlaying) {
          // Start a fresh hand
          if (!state.gameState) {
            get().initializeGame();
          }
          get().startHand();
        }
      },

      setAutoPlaySpeed: (speed) => {
        set({ autoPlaySpeed: speed });
      },

      recordHumanDecision: (decision) => {
        const state = get();
        const entry: BotEvaluationEntry = {
          ...decision,
          handNumber: state.gameState?.handNumber || 0,
        };
        set({ botEvaluations: [...state.botEvaluations, entry] });
      },
    }),
    {
      name: 'pokerbot-arena-storage',
      partialize: (state) => ({
        stats: state.stats,
        currentBankroll: state.currentBankroll,
        gameHistory: state.gameHistory,
        botEvaluations: state.botEvaluations,
        selectedBotPersonality: state.selectedBotPersonality,
        tableSize: state.tableSize,
        buyIn: state.buyIn,
        blinds: state.blinds,
        autoPlaySpeed: state.autoPlaySpeed,
      }),
    }
  )
);

// Helper: estimate if human made the right decision
function calculateHumanAccuracy(gameState: GameState | null, _handNumber: number): number {
  if (!gameState) return 50;
  
  const human = gameState.players.find(p => p.id === 'human');
  if (!human || human.folded) return 50;
  
  const handEval = evaluateHand(human.hand, gameState.communityCards);
  const strength = handEval.score / 9500000;
  const toCall = gameState.currentBet - human.bet;
  
  // Simple heuristic: if strength > pot odds, call/raise is correct; if not, fold/check is correct
  const potOdds = toCall > 0 ? toCall / (gameState.pot + toCall) : 0;
  
  if (strength > potOdds) {
    // Should call or raise
    return human.bet >= gameState.currentBet ? 85 : 90; // check or bet = good
  } else {
    // Should fold or check
    return human.folded ? 95 : 70; // fold = great, call = bad
  }
}

                <Play size={14} /> {gameState?.gameOver ? 'New Game' : 'Start'}
              </button>
            )}
          </div>

          {/* Tab Navigation */}
          <nav className="flex gap-1 bg-surface-elevated rounded-lg p-1 overflow-x-auto">
            {tabs.map(tab => (
              <button
                key={tab.id}
                onClick={() => setActiveTab(tab.id)}
                className={`flex items-center gap-1.5 px-3 py-1.5 rounded-md text-sm font-medium transition-all whitespace-nowrap ${
                  activeTab === tab.id
                    ? 'bg-gold text-black shadow-md'
                    : 'text-text-secondary hover:text-text-primary'
                }`}
              >
                <tab.icon size={14} />
                <span>{tab.label}</span>
              </button>
            ))}
          </nav>
        </div>
      </header>

      {/* Main Content */}
      <main className="max-w-7xl mx-auto px-4 py-4">
        {/* Game Status Bar */}
        {activeTab === 'play' && isPlaying && gameState && (
          <div className="flex items-center justify-between bg-surface-elevated rounded-xl px-4 py-2.5 border border-surface-border mb-4">
            <div className="flex items-center gap-6 text-sm">
              <span className="text-text-secondary">
                Hand <span className="text-text-primary font-mono font-semibold">{gameState.handNumber}</span>
              </span>
              <span className="text-text-secondary">
                Blinds <span className="text-text-primary font-mono font-semibold">${useGameStore(s => s.blinds.small)}/${useGameStore(s => s.blinds.big)}</span>
              </span>
              <span className="text-text-secondary">
                Phase <span className="text-gold font-mono font-semibold uppercase">{gameState.currentPhase}</span>
              </span>
            </div>
            <div className="flex items-center gap-4">
              <span className="text-text-secondary">
                Bankroll <span className="text-gold font-mono font-bold text-base">${useGameStore(s => s.currentBankroll)}</span>
              </span>
              {gameState.gameOver && (
                <button onClick={handleNextHand} className="btn-primary flex items-center gap-1.5 px-3 py-1.5 rounded-lg text-xs font-semibold">
                  <Play size={12} /> Next Hand
                </button>
              )}
            </div>
          </div>
        )}

        {activeTab === 'play' && (
          <div className="space-y-4">
            <PokerTable />
            <RiskOverlay />
            <PlayerControls />
          </div>
        )}

        {activeTab === 'stats' && <StatsDashboard />}
        {activeTab === 'settings' && <SettingsPanel />}
        {activeTab === 'history' && <HandHistoryPage />}
        {activeTab === 'leaderboard' && <LeaderboardPage />}
      </main>

      {/* Footer */}
      <footer className="border-t border-surface-border mt-8 py-4 text-center">
        <p className="text-xs text-text-secondary">
          PokerBot Arena — Educational simulation. No real money involved.
        </p>
      </footer>
    </div>
  );
};

const LeaderboardPage: React.FC = () => {

        <Brain size={18} className="text-gold" />

                <span className="font-mono">{bot.mistakeRate}%</span>
              </div>

            </div>
          </div>
        ))}

          Overall Performance
        </h3>
        <div className="grid grid-cols-2 md:grid-cols-4 gap-3 text-sm">

            <div className="text-xl font-bold font-mono">{stats.totalHands}</div>

};
